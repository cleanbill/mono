7. Click away

  <body>
      I can dance for <b name='danceFor' ></b> hours

      <button click='addHour' >Longer</button>
      <button click='takeHour' >Less</button>
  </body>
  
  <script type='module'>
      import { go, tools, addClickFunction } from "./dist/binder.js";
      go();

      tools.setByName("danceFor", 0);

      const changeHours = amount => {
        const hours = parseInt(tools.getByName("danceFor"));
        tools.setByName("danceFor", hours+amount);
      }
      addClickFunction("addHour",(tools, ev) => changeHours(1));
      addClickFunction("takeHour",(tools, ev) => changeHours(-1));
  </script>
I can dance for hours
